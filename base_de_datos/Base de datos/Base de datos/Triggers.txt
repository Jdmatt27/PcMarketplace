
CREATE OR REPLACE TRIGGER trg_clientes_bi
    BEFORE INSERT ON clientes
    FOR EACH ROW
BEGIN
    IF :NEW.id_cliente IS NULL THEN
        :NEW.id_cliente := seq_clientes.NEXTVAL;
    END IF;
END;
/
